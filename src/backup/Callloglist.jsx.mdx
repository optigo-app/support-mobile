import {
  Box,
  Typography,
  Avatar,
  Chip,
  ListItemButton,
  List,
  Divider
} from "@mui/material";

  CallMissedRounded,
  CallReceivedRounded,
  CallMadeRounded,
  PhoneCallbackRounded
} from "@mui/icons-material";

import { EmailScrollArea } from "../../ui/ScrollArea";
import TicketFilterDrawer from "../Tickets/TicketFilterDrawer";
import PremiumFloatingPagination from "../../ui/PremiumFloatingPagination";
import GmailStyleHeader from "./Header"; // Adjust path if needed
import CallLogDetailPage from "./Details"; // Adjust path if needed

// --- 1. CONSTANTS & UTILS ---
const COLORS = {
  textPrimary: "#1A1A1A",
  textSecondary: "#777777",
  // Status Colors
  missedBg: "rgba(220, 38, 38, 0.08)",
  missedText: "#dc2626",
  incomingBg: "rgba(74, 102, 255, 0.08)",
  incomingText: "#4A66FF",
  outgoingBg: "rgba(58, 248, 126, 0.08)",
  outgoingText: "#059669",
  neutralBg: "rgba(0, 0, 0, 0.06)",
  neutralText: "#555",
};

function getAvatarGradient(name) {
  const gradients = [
    "linear-gradient(135deg, #FF9966 0%, #FF5E62 100%)",
    "linear-gradient(135deg, #56CCF2 0%, #2F80ED 100%)",
    "linear-gradient(135deg, #11998e 0%, #38ef7d 100%)",
    "linear-gradient(135deg, #8E2DE2 0%, #4A00E0 100%)",
    "linear-gradient(135deg, #f093fb 0%, #f5576c 100%)"
  ];
  const index = name ? name.length % gradients.length : 0;
  return gradients[index];
}

function getStatusStyle(status) {
  const normalized = status?.toLowerCase() || "";
  if (normalized === "missed") {
    return { bg: COLORS.missedBg, color: COLORS.missedText, icon: <CallMissedRounded sx={{ fontSize: 12 }} /> };
  } else if (normalized === "incoming") {
    return { bg: COLORS.incomingBg, color: COLORS.incomingText, icon: <CallReceivedRounded sx={{ fontSize: 12 }} /> };
  } else if (normalized === "outgoing") {
    return { bg: COLORS.outgoingBg, color: COLORS.outgoingText, icon: <CallMadeRounded sx={{ fontSize: 12 }} /> };
  }
  return { bg: COLORS.neutralBg, color: COLORS.neutralText, icon: <PhoneCallbackRounded sx={{ fontSize: 12 }} /> };
}

// --- 2. MOCK DATA ---
const generateMockData = () => {
  const companies = ["TechFlow", "MediCare", "LogiStick", "FinCorp", "EduLearn"];
  const callers = ["Mira Patel", "Rushi Patel", "Diaa", "Khushi", "Admin"];

  return Array.from({ length: 25 }).map((_, i) => {
    const isMissed = Math.random() < 0.2;
    return {
      sr: i,
      company: companies[i % companies.length],
      callBy: callers[i % callers.length],
      appName: "Sales CRM",
      description: isMissed 
        ? "Caller hung up before connection." 
        : `Discussion regarding policy update #${100 + i}`,
      date: `2${i % 9}/11/2025`,
      time: `${10 + (i % 12)}:${10 + (i % 49)} ${i % 2 === 0 ? 'AM' : 'PM'}`,
      externalStatus: isMissed ? "Missed" : i % 2 === 0 ? "Incoming" : "Outgoing",
      rating: isMissed ? 0 : 4,
      attendBy: "Admin",
    };
  });
};

const callLogsList = generateMockData();

// --- 3. COMPONENT ---
const CallLogsApp = () => {
  const [searchQuery, setSearchQuery] = useState("");
  const [page, setPage] = useState(0);
  const [rowsPerPage, setRowsPerPage] = useState(20);
  const [anchorElSort, setAnchorElSort] = useState(false);
  
  // Detail Drawer States
  const [open, setOpen] = useState(false);
  const [selectedLog, setSelectedLog] = useState(null);
  
  const scrollRef = useRef(null);

  // Filter Logic
  const filteredLogs = useMemo(() => {
    return callLogsList.filter((log) => {
      const displayTitle = log.company;
      return displayTitle.toLowerCase().includes(searchQuery.toLowerCase()) || 
             log.callBy.toLowerCase().includes(searchQuery.toLowerCase());
    });
  }, [searchQuery]);

  const paginatedLogs = useMemo(() => {
    return filteredLogs.slice(page * rowsPerPage, page * rowsPerPage + rowsPerPage);
  }, [filteredLogs, page, rowsPerPage]);

  const Scroller = () => {
    if (scrollRef.current) {
      scrollRef.current.scrollTo({ top: 0, behavior: "smooth" });
    }
  };

  const handleChangePage = (event, newPage) => {
    Scroller();
    setPage(newPage);
  };

  const handleChangeRowsPerPage = (event) => {
    Scroller();
    setRowsPerPage(parseInt(event.target.value, 10));
    setPage(0);
  };

  const handleOpenDrawer = (log) => {
    setSelectedLog(log);
    setOpen(true);
  };

  const handleCloseDrawer = () => {
    setOpen(false);
    setSelectedLog(null);
  };

  return (
    <Box sx={{ display: "flex", flexDirection: "column", overflow: "hidden", bgcolor: "#fff", height: "100%" }}>
      
      {/* HEADER */}
      <GmailStyleHeader 
        anchorElSort={anchorElSort} 
        setAnchorElSort={setAnchorElSort} 
        searchQuery={searchQuery} 
        setSearchQuery={setSearchQuery} // Pass string updater if needed, or event handler
        onSearch={(e) => setSearchQuery(e.target.value)}
        title="Call Logs" 
      />

      {/* SCROLL AREA */}
      <EmailScrollArea ref={scrollRef}>
        {paginatedLogs.length === 0 ? (
          <Box sx={{ p: 5, textAlign: "center", color: COLORS.textSecondary }}>
            <Typography variant="body2">No logs found</Typography>
          </Box>
        ) : (
          <List disablePadding>
            {paginatedLogs.map((log) => {
              // --- MAPPING LOGIC (Matching Reference) ---
              const statusStyle = getStatusStyle(log.externalStatus);
              const displayTitle = log.company;
              
              return (
                <React.Fragment key={log.sr}>
                  <ListItemButton
                    onClick={() => handleOpenDrawer(log)}
                    sx={{
                      py: 1.2, // Reference padding
                      px: 2,
                      display: "flex",
                      alignItems: "flex-start",
                      gap: 2,
                      borderBottom: "1px solid rgba(0,0,0,0.04)",
                      transition: "0.2s",
                      bgcolor: "#fff",
                      "&:hover": {
                        bgcolor: "rgba(0,0,0,0.02)",
                      },
                    }}
                  >
                    {/* 1. LEFT SIDE: Avatar */}
                    <Avatar
                      sx={{
                        width: 40,
                        height: 40,
                        borderRadius: "50%",
                        fontWeight: 700,
                        fontSize: "1rem",
                        background: getAvatarGradient(displayTitle),
                        color: "#fff",
                        mt: 0.5,
                      }}
                    >
                      {displayTitle ? displayTitle.charAt(0).toUpperCase() : "C"}
                    </Avatar>

                    {/* 2. RIGHT SIDE: Main Content */}
                    <Box sx={{ flex: 1, minWidth: 0 }}>
                      
                      {/* Row 1: Title & Time */}
                      <Box sx={{ display: "flex", justifyContent: "space-between", alignItems: "center", mb: 0.3 }}>
                        <Typography
                          sx={{
                            fontWeight: 700,
                            fontSize: "0.9rem",
                            color: COLORS.textPrimary,
                            overflow: "hidden",
                            textOverflow: "ellipsis",
                            whiteSpace: "nowrap",
                          }}
                        >
                          {displayTitle}
                        </Typography>

                        {/* Time aligned to right */}
                        <Typography sx={{ fontSize: "0.75rem", color: COLORS.textSecondary, whiteSpace: "nowrap", ml: 1 }}>
                          {log.time}
                        </Typography>
                      </Box>

                      {/* Row 2: Status + Description Snippet */}
                      <Box sx={{ display: "flex", alignItems: "center", gap: 1, flexWrap: "nowrap" }}>
                        
                        {/* Status Chip */}
                        <Chip
                          label={log.externalStatus}
                          size="small"
                          icon={statusStyle.icon}
                          sx={{
                            height: 18,
                            fontSize: "0.65rem",
                            fontWeight: 700,
                            bgcolor: statusStyle.bg,
                            color: statusStyle.color,
                            borderRadius: "4px",
                            px: 0.5,
                            "& .MuiChip-label": { padding: "0 4px" },
                            "& .MuiChip-icon": { color: "inherit", fontSize: "0.9rem", ml: -0.5 }
                          }}
                        />

                        {/* Description Line */}
                        <Typography
                          sx={{
                            fontSize: "0.8rem",
                            color: COLORS.textSecondary,
                            overflow: "hidden",
                            textOverflow: "ellipsis",
                            whiteSpace: "nowrap",
                            flex: 1,
                          }}
                        >
                          <span style={{ color: "#ccc", marginRight: "6px" }}>|</span>
                          <span style={{ fontWeight: 500, color: "#555" }}>{log.callBy}</span>
                          <span style={{ margin: "0 4px" }}>-</span>
                          {log.description}
                        </Typography>
                      </Box>

                    </Box>
                  </ListItemButton>
                </React.Fragment>
              );
            })}
          </List>
        )}
      </EmailScrollArea>

      {/* FOOTER & OVERLAYS */}
      <PremiumFloatingPagination 
        page={page} 
        rowsPerPage={rowsPerPage} 
        count={filteredLogs.length} 
        onPageChange={handleChangePage} 
        onRowsPerPageChange={handleChangeRowsPerPage} 
      />

      <TicketFilterDrawer 
        onClose={() => setAnchorElSort(false)} 
        open={anchorElSort} 
        title="Filter Logs" 
      />

      {/* Passing the specific log data here */}
      <CallLogDetailPage 
        open={open} 
        onClose={handleCloseDrawer} 
        onBack={handleCloseDrawer} 
        logData={selectedLog} 
      />
    </Box>
  );
};

export default CallLogsApp;
