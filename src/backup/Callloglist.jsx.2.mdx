  Box,
  Typography,

  List,
  ListItemButton,
  ListItemAvatar,
  Divider,
  useTheme,
} from "@mui/material";

  CallMissedRounded,
  CallReceivedRounded,
  CallMadeRounded,
} from "@mui/icons-material";
import ScrollArea, { EmailScrollArea } from "../../ui/ScrollArea";
import PremiumFloatingPagination from "../../ui/PremiumFloatingPagination";
import GmailStyleHeader from "./Header";
import CallLogDetailPage from "./Details";

// --- 1. MOCK DATA ---
const generateMockData = () => {
  const companies = ["TechFlow Systems", "MediCare Plus", "LogiStick Inc", "FinCorp Global", "EduLearn"];
  const names = ["Mira Patel", "Rushi Patel", "Diaa", "Khushi", "Admin Hub"];
  
  return Array.from({ length: 25 }).map((_, i) => {
    const isMissed = Math.random() < 0.2;
    return {
      sr: i,
      date: `2${i % 9}/11/2025`,
      time: `${10 + (i % 12)}:${10 + (i % 49)}`,
      company: companies[i % companies.length],
      callBy: names[i % names.length],
      attendBy: names[(i + 1) % names.length],
      appName: "Sales CRM",
      description: isMissed
        ? "Missed call."
        : `Discussion regarding the Q3 maintenance tasks and feedback loop #${400+i}.`,
      externalStatus: isMissed ? "Missed" : "Incoming",
      rating: isMissed ? 0 : (i % 5) + 1,
      feedback: isMissed ? "" : i % 3 === 0 ? "Excellent support." : "",
    };
  });
};

const callLogsList = generateMockData();

const COLORS = {
  textPrimary: "#1A1A1A",
  textSecondary: "#64748B",
  missed: "#DC2626",
};

const CallLogsApp = () => {
  // State
  const [searchQuery, setSearchQuery] = useState("");
  const [page, setPage] = useState(0);
  const [rowsPerPage, setRowsPerPage] = useState(15); // Default more rows for compact view
  const [anchorElSort, setAnchorElSort] = useState(false);
  const [isDetailOpen, setIsDetailOpen] = useState(false);
  const [selectedLog, setSelectedLog] = useState(null);
  const scrollRef = useRef(null);

  // Logic
  const filteredLogs = useMemo(() => {
    return callLogsList.filter((log) => 
      log.company.toLowerCase().includes(searchQuery.toLowerCase()) ||
      log.callBy.toLowerCase().includes(searchQuery.toLowerCase())
    );
  }, [searchQuery]);

  const paginatedLogs = useMemo(() => {
    return filteredLogs.slice(page * rowsPerPage, page * rowsPerPage + rowsPerPage);
  }, [filteredLogs, page, rowsPerPage]);

  const handleLogClick = (log) => {
    setSelectedLog(log);
    setIsDetailOpen(true);
  };

  const getAvatarColor = (name) => {
    const colors = ["#4A66FF", "#0891b2", "#be185d", "#d97706", "#7c3aed"];
    return colors[name.length % colors.length];
  };

  const formatTime = (timeStr) => {
    if (!timeStr) return "";
    const [h, m] = timeStr.split(":");
    const hour = parseInt(h, 10);
    const ampm = hour >= 12 ? "PM" : "AM";
    return `${hour % 12 || 12}:${m} ${ampm}`;
  };

  return (
    <Box sx={{ display: "flex", flexDirection: "column", height: "100%", bgcolor: "#fff", overflow: "hidden" }}>
      
      {/* HEADER */}
      <GmailStyleHeader
        anchorElSort={anchorElSort}
        setAnchorElSort={setAnchorElSort}
        title="Call Logs"
        onSearch={(e) => setSearchQuery(e.target.value)}
      />

      {/* SCROLL AREA */}
      <EmailScrollArea ref={scrollRef}>
        <List disablePadding>
          {paginatedLogs.map((log) => {
            const isMissed = log.externalStatus === "Missed";

            return (
              <React.Fragment key={log.sr}>
                <ListItemButton
                  onClick={() => handleLogClick(log)}
                  sx={{
                    py: 1, // Compact Padding (8px)
                    px: 2,
                    alignItems: "flex-start",
                    transition: "0.2s",
                    "&:hover": { bgcolor: "#F3F4F6" },
                  }}
                >
                  {/* COMPACT AVATAR */}
                  <ListItemAvatar sx={{ minWidth: 48, mt: 0.3 }}>
                    <Avatar
                      sx={{
                        width: 36, // Smaller Avatar
                        height: 36,
                        bgcolor: isMissed ? "#FEF2F2" : getAvatarColor(log.company),
                        color: isMissed ? "#DC2626" : "#fff",
                        fontSize: 14,
                        fontWeight: 700,
                      }}
                    >
                      {isMissed ? <CallMissedRounded fontSize="small" /> : log.company.charAt(0)}
                    </Avatar>
                  </ListItemAvatar>

                  {/* DENSE CONTENT */}
                  <Box sx={{ flex: 1, minWidth: 0 }}>
                    {/* Top Row: Company & Time */}
                    <Box display="flex" justifyContent="space-between" alignItems="baseline">
                      <Typography
                        variant="body2"
                        sx={{
                          fontWeight: isMissed ? 700 : 600,
                          color: isMissed ? COLORS.missed : COLORS.textPrimary,
                          fontSize: "0.9rem",
                          whiteSpace: "nowrap",
                          overflow: "hidden",
                          textOverflow: "ellipsis",
                          mr: 1
                        }}
                      >
                        {log.company}
                      </Typography>

                      <Typography
                        variant="caption"
                        sx={{
                          color: isMissed ? COLORS.missed : "#94a3b8",
                          fontWeight: isMissed ? 600 : 400,
                          fontSize: "0.7rem",
                          flexShrink: 0
                        }}
                      >
                        {formatTime(log.time)}
                      </Typography>
                    </Box>

                    {/* Bottom Row: Snippet (Combined Caller + Desc) */}
                    <Typography
                      variant="body2"
                      sx={{
                        color: COLORS.textSecondary,
                        fontSize: "0.8rem",
                        lineHeight: 1.3,
                        overflow: "hidden",
                        whiteSpace: "nowrap", // Force 1 line for maximum compactness
                        textOverflow: "ellipsis",
                      }}
                    >
                      <span style={{ color: "#334155", fontWeight: 500 }}>
                        {log.callBy}
                      </span>
                      {" — "}
                      {log.description}
                    </Typography>
                  </Box>
                </ListItemButton>
                
                {/* Inset Divider (Aligned with text) */}
                <Divider component="li" sx={{ ml: 8, opacity: 0.5, borderColor: "rgba(0,0,0,0.04)" }} />
              </React.Fragment>
            );
          })}
        </List>

        {paginatedLogs.length === 0 && (
          <Box p={4} textAlign="center" color="text.secondary">
            No logs found
          </Box>
        )}
      </EmailScrollArea>

      {/* FOOTER & DRAWERS */}
      <PremiumFloatingPagination
        page={page}
        rowsPerPage={rowsPerPage}
        count={filteredLogs.length}
        onPageChange={(e, p) => setPage(p)}
        onRowsPerPageChange={(e) => { setRowsPerPage(parseInt(e.target.value, 10)); setPage(0); }}
      />

      <TicketFilterDrawer
        onClose={() => setAnchorElSort(false)}
        open={anchorElSort}
        title="Filter Logs"
      />

      {/* Details Drawer with Data Passing */}
      <CallLogDetailPage
        open={isDetailOpen}
        onClose={() => setIsDetailOpen(false)}
        onBack={() => setIsDetailOpen(false)}
        logData={selectedLog}
      />
    </Box>
  );
};

export default CallLogsApp;
